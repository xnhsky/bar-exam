### 3-2. 派生色変数の存在検査関数（S88 詳細）

```python
import re

DERIVATIVE_COLORS = [
    '--accent-light', '--accent-darker',
    '--mid-warm', '--mid-cool',
    '--accent-soft-2', '--mid-soft', '--surface-tint',
    '--neutral-cream', '--contrast-warm', '--contrast-cool',
]

def check_s88_palette_derivatives(html_content: str) -> Optional[str]:
    """S88: 派生色変数存在検査"""
    # :root ブロックを抽出
    root_blocks = re.findall(r':root\s*\{([^}]+)\}', html_content)
    if not root_blocks:
        return "S88 violation: no :root block found"

    # 全 :root ブロック内のテキスト連結
    all_root_content = '\n'.join(root_blocks)

    # 10 派生色変数の存在チェック
    missing = []
    for var in DERIVATIVE_COLORS:
        # var: #xxxxxx の hex 形式チェック
        pattern = rf'{re.escape(var)}\s*:\s*#[0-9a-fA-F]{{6}}'
        if not re.search(pattern, all_root_content):
            missing.append(var)

    if missing:
        return f"S88 violation (AP-45): missing derivative colors: {', '.join(missing)}"

    return None
```

### 3-3. メタ説明検査関数（S90 詳細）

```python
META_EXPLANATION_PATTERNS = [
    # 肢系
    r'肢[\d１-５]\s*を選[ぶびうい]',
    r'肢[\d１-５]\s*を選択',
    r'肢[\d１-５]\s*が正解',
    r'正解の肢は',
    r'本問の正解は',
    # 記号系
    r'記号[ア-オ]\s*を選[ぶびうい]',
    r'[ア-オ]の組合せが正解',
    # 手順系
    r'解答の手順',
    r'解答プロセス',
    r'正解を選ぶプロセス',
    r'解答に至る手順',
    # メタ説明系
    r'本問では.+を選[べびう]',
]

META_CHECK_SCOPES = [
    '.basis-card-body',
    '.sub-card.professor',
    '.sub-card.explanation',
    'section#c-4',
    'section#mindmap-tree text',
    'section#mindmap-radial text',
    'section#c-5 text',
    '.key-phrase-box',
]

META_EXEMPT_SCOPES = [
    '.answer-instruction',
    '.fa-summary',
    'section#mindmap-tree .issue-fill text',  # 本問の論点枠
    'section#mindmap-radial .issue-branch-fill text',
]

def check_s90_meta_explanation(html_content: str) -> Optional[str]:
    """S90: メタ説明違反検査"""
    from bs4 import BeautifulSoup
    soup = BeautifulSoup(html_content, 'html.parser')

    violations = []
    for scope_selector in META_CHECK_SCOPES:
        elements = soup.select(scope_selector)
        for elem in elements:
            # 例外スコープに該当しないか確認
            if any(elem.find_parent(exempt) for exempt in META_EXEMPT_SCOPES):
                continue
            text = elem.get_text()
            for pattern in META_EXPLANATION_PATTERNS:
                if re.search(pattern, text):
                    violations.append(f"  {scope_selector}: {pattern} matched in: {text[:80]}")

    if violations:
        return f"S90 violation (AP-43): meta-explanation detected\n" + '\n'.join(violations)
    return None
```

### 3-4. 教授解説密度検査関数（S91 詳細）

```python
PROF_DENSITY_REQUIREMENTS = {
    '.prof-point': 150,
    '.prof-process': 400,
    '.prof-image': 300,
    '.prof-application': 300,
}

def count_text_chars(elem):
    """要素内のテキスト文字数（タグ除去後）"""
    return len(elem.get_text().strip())

def check_s91_professor_density(html_content: str) -> Optional[str]:
    """S91: 教授解説密度検査"""
    from bs4 import BeautifulSoup
    soup = BeautifulSoup(html_content, 'html.parser')

    violations = []
    profs = soup.select('.sub-card.professor')
    for prof in profs:
        for selector, min_chars in PROF_DENSITY_REQUIREMENTS.items():
            heading = prof.select_one(selector)
            if heading is None:
                violations.append(f"  missing element: {selector}")
                continue
            char_count = count_text_chars(heading)
            if char_count < min_chars:
                violations.append(
                    f"  {selector}: {char_count} chars (required: ≥{min_chars})"
                )

        # prof-image の 3 要素存在チェック
        prof_image = prof.select_one('.prof-image')
        if prof_image:
            for sub in ['.image-scene', '.image-bridge', '.image-contrast']:
                if not prof_image.select_one(sub):
                    violations.append(f"  prof-image missing: {sub}")

        # syllogism 3 要素存在チェック
        syllogism = prof.select_one('.syllogism')
        if syllogism:
            for sub in ['.syl-major', '.syl-minor', '.syl-conclusion']:
                if not syllogism.select_one(sub):
                    violations.append(f"  syllogism missing: {sub}")

    if violations:
        return f"S91 violation (AP-44): professor-density-v2 insufficient\n" + '\n'.join(violations)
    return None
```

# 既存 -SpecVersion パラメータ追加
param(
    [string]$SpecVersion = 'v9.2.0',
    [int]$MaxProblems = 5,
    [int]$MaxConsecutiveFailures = 3,
    [switch]$DryRun
)

# spec ファイル選択
$specFile = switch ($SpecVersion) {
    'v9.2.0' { 'spec/tx-v9.2.0-deepdive-core.md' }
    'v9.1.0' { 'spec/tx-v9.1.0-mindmap-core.md' }
    default { throw "Unknown spec version: $SpecVersion" }
}

# 各問生成後の validate-tx 実行
foreach ($problem in $problems) {
    $htmlPath = Generate-HTML -ProblemJson $problem -SpecFile $specFile
    $validateResult = python scripts/validate-tx.py $htmlPath
    if ($LASTEXITCODE -ne 0) {
        Write-Warning "Validation failed for $htmlPath"
        $consecutiveFailures++
        if ($consecutiveFailures -ge $MaxConsecutiveFailures) {
            Write-Error "Aborting: $MaxConsecutiveFailures consecutive failures"
            break
        }
    } else {
        $consecutiveFailures = 0
    }
}
