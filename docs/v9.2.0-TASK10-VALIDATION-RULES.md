def detect_spec_version(html_content: str) -> str:
    """footer-spec の feature-tag から spec バージョンを判定"""
    if 'TX v9.2.0 DEEP-DIVE' in html_content:
        return 'v9.2.0'
    if 'TX v9.1.0 MINDMAP' in html_content:
        return 'v9.1.0'
    if 'TX v9.0.0 GENKEI' in html_content:
        return 'v9.0.0'
    if 'TX v8.11.7' in html_content:
        return 'v8.11.7'
    return 'unknown'
